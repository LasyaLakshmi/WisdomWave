21. Structures and Unions

// Structure definition
struct Person {
  char name[50];
  int age;
};

struct Person person1 = {"John", 25};

// Union definition
union Data {
  int i;
  float f;
  char str[20];
};

union Data data1;
data1.i = 10;  // Only one member can hold a value at a time

22. File Handling

FILE *file = fopen("example.txt", "w");
if (file != NULL) {
  fprintf(file, "Hello, world!");
  fclose(file);
}

FILE *readFile = fopen("example.txt", "r");
if (readFile != NULL) {
  char str[100];
  fscanf(readFile, "%s", str);
  printf("%s\n", str);
  fclose(readFile);
}

23. Command Line Arguments

#include <stdio.h>

int main(int argc, char *argv[]) {
  if (argc > 1) {
    printf("Hello, %s!\n", argv[1]);  // argv[1] is the first command-line argument
  } else {
    printf("No arguments passed.\n");
  }
  return 0;
}

24. Typedef and Enum

// Typedef
typedef unsigned long ulong;
ulong num = 1000;

// Enum
enum Day {Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday};
enum Day today = Wednesday;
printf("Today is %d\n", today);  // Prints 2
