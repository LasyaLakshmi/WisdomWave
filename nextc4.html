18. Dynamic Memory Allocation

int *arr = (int*) malloc(5 * sizeof(int));  // Allocating memory for array
if (arr == NULL) {
  printf("Memory allocation failed!\n");
  return 1;
}
for (int i = 0; i < 5; i++) {
  arr[i] = i + 1;
  printf("%d ", arr[i]);
}
free(arr);  // Freeing allocated memory

19. Storage Classes

// Automatic (default for local variables)
int autoVar = 5;

// Static (value persists across function calls)
void example() {
  static int count = 0;
  count++;
  printf("%d\n", count);
}

// External (variable defined outside functions)
extern int globalVar;

void demo() {
  example();
  example();
}
