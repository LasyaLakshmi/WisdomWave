15. switch Statement

int day = 3;
switch(day) {
  case 1: printf("Monday"); break;
  case 2: printf("Tuesday"); break;
  case 3: printf("Wednesday"); break;
  default: printf("Other day");
}

16. Loops: while, for, do-while

int i = 0;
while (i < 3) {
  printf("%d ", i);
  i++;
}

for (int j = 0; j < 3; j++) {
  printf("%d ", j);
}

i = 0;
do {
  printf("%d ", i);
  i++;
} while (i < 3);
